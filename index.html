require 'json'
require 'open-uri'

def fetch_exchange_rates(base_currency)
url = "https://api.exchangeratesapi.io/latest?base=#{base_currency}"
response = URI.open(url).read
exchange_rates = JSON.parse(response)
return exchange_rates['rates']
end

def convert_currency(amount, from_currency, to_currency)
rates = fetch_exchange_rates(from_currency)
if rates.key?(to_currency)
converted_amount = amount * rates[to_currency]
return converted_amount.round(2)
else
puts "Conversion rate not available for #{to_currency}."
return nil
end
end

def main
puts "Welcome to Currency Converter"
print "Enter the amount: "
amount = gets.chomp.to_f

print "Enter the currency you're converting from (e.g., USD, EUR, GBP): "
from_currency = gets.chomp.upcase

print "Enter the currency you're converting to (e.g., USD, EUR, GBP): "
to_currency = gets.chomp.upcase

converted_amount = convert_currency(amount, from_currency, to_currency)
if converted_amount
puts "#{amount} #{from_currency} is equivalent to #{converted_amount} #{to_currency}."
else
puts "Failed to convert currency. Please try again."
end
end
